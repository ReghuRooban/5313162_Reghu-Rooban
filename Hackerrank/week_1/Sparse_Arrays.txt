#include <stdio.h>
#include <string.h>

#define MAX_STRINGS 1000
#define MAX_STRING_LENGTH 101

int main() {
    int n, q;
    char strings[MAX_STRINGS][MAX_STRING_LENGTH];
    char queries[MAX_STRINGS][MAX_STRING_LENGTH];
    int result[MAX_STRINGS] = {0};
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        scanf("%s", strings[i]);
    }
    scanf("%d", &q);
    for (int i = 0; i < q; i++) {
        scanf("%s", queries[i]);
    }
    for (int i = 0; i < q; i++) {
        for (int j = 0; j < n; j++) {
            if (strcmp(queries[i], strings[j]) == 0) {
                result[i]++;
            }
        }
    }

    for (int i = 0; i < q; i++) {
        printf("%d\n", result[i]);
    }

    return 0;
}
